def render_to_pdf(template_src, context_dict={}):
    template = get_template(template_src)
    html = template.render(context_dict)
    result = BytesIO()
    pdf = pisa.pisaDocument(BytesIO(html.encode("ISO-8859-1")), result)
    if not pdf.err:
        return HttpResponse(result.getvalue(), content_type='application/pdf')
    return None


# Opens up page as PDF
class ViewPDF(View):
    viewPDF = RequestForm.objects.filter(approved="Permitted")
    my_dict = list(viewPDF)
    count = RequestForm.objects.filter(approved="Permitted").count()

    # Get time
    t = time.localtime()
    current_time = time.strftime("%D - %H:%M:%S", t)

    context = {'my_dict': my_dict, 'count': count, 'current_time': current_time}

    def get(self, request, *args, **kwargs):
        pdf = render_to_pdf('pdf/pdf_template.html', self.context)
        return HttpResponse(pdf, content_type='application/pdf')
